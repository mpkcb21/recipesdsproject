recipe_stats["std"] = recipe_stats["std"].fillna(0)
recipe_stats["is_perfect"] = (recipe_stats["mean"] == 5).astype(int)
recipe_stats["dissatisfaction_rate"] = 1 - (recipe_stats["mean"] / 5)
In[]:
def classify(row, mean_threshold=4.0):

    #Perfect = nothing to explain
    if row["mean"] == 5:
        return "Perfect"
    
    #Clear recipe flaw: universally mediocre/bad
    if row["mean"] < mean_threshold:
        return "Recipe Flaw Likely"
    if row['std'] < 1:
        return 'User-Preference/Flaw'
    #Any disagreement = preference or execution issues
    return "Disagreement-Driven"

recipe_stats["issue_type"] = recipe_stats.apply(classify, axis=1)

I created groups of possible reasons why a recipe would not get a 5 by either it being perfect, user-preference/flaw, recipe flaw, and disagreement-driven(basically neither the recipe or users flaw). I did so by agrregating the means and standard deviations of each recipe and assigned proper thresholds to divide the recipes. Basically saying that if the recipe had a higher average rating and the standard deviation was low it was probably the users preference or flaw that they mean of the recipe is not a perfect 5. If its a high mean but also a high standard deviation it implies that there is a high variation and that the high average doesn't really mean that its a user flaw as many people are rating below a 5, which leads to a disagreement driven score for the recipe. I also said that if the recipe has a average score of below 4 it just means that the recipe was flawed as that was scores below 4 are quite rare anyways so standard deviation doesnt matter at that point.

#TF-IDF + Logistic Regression model(log regr with tf idf allows us to do a log regr using text feature and extract emotions written in a review)
# Replace missing review text with empty string
combined["review_text"] = combined["review"].fillna("")

# Past user rating average - subtract the current columns rating so the column we are using is not used to predict itself, basically use user_counts in a more meaningful way
combined = combined.sort_values("date")
combined["past_review_count"] = combined.groupby("user_id").cumcount()
combined["past_rating_sum"] = combined.groupby("user_id")["rating"].cumsum() - combined["rating"]
combined["user_avg_rating_past"] = combined["past_rating_sum"] / combined["past_review_count"]
combined["user_avg_rating_past"] = combined["user_avg_rating_past"].fillna(combined["rating"].mean())

# Split data
X = combined[["review_text", "user_avg_rating_past", "recipe_review_count", ]]
y = combined["good_rating"]

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.25, random_state=42, stratify=y)
 
# Column Transformer
preprocess = ColumnTransformer(transformers=[("tfidf", TfidfVectorizer(max_features=3000, stop_words="english"), "review_text"),("num", "passthrough", ["user_avg_rating_past", "recipe_review_count"])])

# Build a modeling pipeline (preprocessing step (TF-IDF + numeric scaling/encoding))
# logistic Regression classifier with maximum 500 iterations to ensure convergence (gradient ascent)
pipe = Pipeline([("preprocess", preprocess),("clf", LogisticRegression(max_iter=500))])

# Define hyperparameter search space for GridSearchCV, gets the best combination that optimizes the model
# C = inverse regularization strength (penalty which helps with overfitting)
param_grid = {"clf__C": [0.1, 1, 2, 5],"clf__class_weight": [None, "balanced"],}
grid = GridSearchCV(pipe,param_grid,scoring="f1",cv=3,n_jobs=-1,verbose=1)

#fit the model to the train set
grid.fit(X_train, y_train)

#gets the best estimator C to predict if the user gave a good or bad review
y_pred = grid.best_estimator_.predict(X_test)
f1 = f1_score(y_test, y_pred)

print("Best Parameters:", grid.best_params_)
print("Final Model F1 Score:", f1)

f1_low_obs = f1_score(y_test.loc[low_users], y_pred.loc[low_users])

print("Observed F1 (High Activity Users):", round(f1_high_obs, 4))
print("Observed Difference (High - Low):", round(obs_diff, 4))


# permutation test to assess fairness

n_permutations = 1000
perm_diffs = []

for _ in range(n_permutations):
    shuffled = np.random.permutation(X_test_copy["user_review_count"])
    median_shuff = np.median(shuffled)

    low_shuff = X_test_copy.index[shuffled <= median_shuff]
    high_shuff = X_test_copy.index[shuffled > median_shuff]

    f1_low = f1_score(y_test.loc[low_shuff], y_pred.loc[low_shuff])
    f1_high = f1_score(y_test.loc[high_shuff], y_pred.loc[high_shuff])

    perm_diffs.append(f1_high - f1_low)

perm_diffs = np.array(perm_diffs)


p_value = np.mean(perm_diffs >= obs_diff)
